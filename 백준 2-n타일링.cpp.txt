#include <iostream>
#include <queue>
using namespace std;
int *dp;
int go(int N)
{
	if (N <= 3)
	{
		dp[N] = N;
		return N;
	}
	
	if (dp[N] > 0)
		return dp[N];
	else
	{
		dp[N] = (go(N - 1) + go(N - 2))%10007;
		return dp[N];
	}
	
}
int main()
{
	int N;
	cin >> N;
	dp = new int[N+1];
	cout << go(N) << endl;

	return 0;
}
