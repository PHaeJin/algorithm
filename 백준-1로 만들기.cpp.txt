#include <iostream>
#include <queue>
using namespace std;
int *dp;
int go(int N)
{
	if (N <= 1)
	{
		dp[N] = 0;
		return 0;
	}
	else if (N <= 3)
	{
		dp[N] = 1;
		return 1;
	}
	if (dp[N] > 0)
	{
		return dp[N];
	}
	else
	{
		bool check = false;
		int num = -1;
		if (N % 3 == 0)
		{
			check = true;
			dp[N] = go(N/3) + 1;
		}
		if (N % 2 == 0)
		{
			if (check)
			{
				num = go(N / 2) + 1;
				if (dp[N] > num)
				{
					dp[N] = num;
				}
			}
			else
			{
				check = true;
				dp[N] = go(N / 2) + 1;
			}
		}
		if (check)
		{
			num = go(N - 1) + 1;
			if (dp[N] > num)
			{
				dp[N] = num;
			}
		}
		else 
		{
			dp[N] = go(N - 1) + 1;
		}
		
	}
	return dp[N];
}
int main()
{
	int N,result;
	cin >> N;

	dp = new int[N+1];
	
	result=go(N);

	cout << result << endl;

	return 0;
}
